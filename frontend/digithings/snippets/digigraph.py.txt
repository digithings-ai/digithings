@dataclass
class ToolContext:
    """Execution context passed to every orchestrator tool handler."""
    session_id: str | None
    run_data_dir: str | None
    index_name: str
    index_config: dict[str, Any]
    state: dict[str, Any]
    allowed_tool_names: frozenset[str] | None = None
    request_id: str | None = None
    workflow_id: str | None = None

def register_tool(name, schema, handler, tags=None, schema_factory=None):
    """Register an orchestrator tool. Schema must be OpenAI function tool format."""
    if name in _tools:
        raise ValueError(f"Orchestrator tool {name!r} is already registered")
    _tools[name] = (dict(schema) if schema else None,
                    schema_factory, handler, set(tags or []))
