export async function POST(req: Request) {
  const machine = await validateMachineApiKey(
    req.headers.get("authorization"));
  const session = await auth();
  if (!machine && !session?.user) return unauthorized();

  const { upstreamBearer, litellmProxyApiKey } =
    await resolveDigigraphUpstreamAuth(req, tenantSlug, ownerUserSub);

  // BYOK: forward per-request key to DigiGraph; never log or persist
  const byokKey = req.headers.get("x-byok-key")?.trim() ?? "";
  if (byokKey) upstreamHeaders["X-BYOK-Key"] = byokKey;

  const provider = createDigiGraphClient(eco.digigraphUrl, upstreamBearer);
  const result = streamText({
    model: provider(digigraphModelName()),
    messages: coreMessages,
    headers: upstreamHeaders,
    experimental_transform: smoothStream({ chunking: "word" }),
  });
  return result.toUIMessageStreamResponse({ headers: responseHeaders });
}
