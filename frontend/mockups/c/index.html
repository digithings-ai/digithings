// Act I — DigiGraph

Route any request through the right specialist agent.

// Supervisor graph dispatches
// to sub-graphs via MCP tools.
// Checkpointed. Tool-discovered.
// OpenAI-compatible surface.

  • // langgraph.supervisor + sub-graphs
  • // MCP tool registry (builtin + vertical)
  • // LiteLLM routing w/ caching
from langgraph.graph import StateGraph, END
from digigraph.orchestration import registry, router

def build_supervisor() -> StateGraph:
    g = StateGraph(AgentState)
    g.add_node("router", router.route)
    for tool in registry.discover():
        g.add_node(tool.name, tool.run)
        g.add_edge("router", tool.name)
        g.add_edge(tool.name, END)
    g.set_entry_point("router")
    return g.compile(checkpointer="sqlite")

# supervisor routes to digiquant, digisearch, digichat…
supervisor = build_supervisor()
result = supervisor.invoke({"prompt": user_input})

// Act II — DigiQuant

Backtest, optimize, and deploy trading strategies.

// Atlas — research & discovery.
// Hermes — signals & alerts.
// Kairos — execution & scheduling.
// All on NautilusTrader.

  • // NautilusTrader engine
  • // Polars dataframes (no pandas)
  • // Strategy registry + versioning
from nautilus_trader.backtest.engine import BacktestEngine
from digiquant.atlas import StrategyRegistry

engine = BacktestEngine(venue="NASDAQ")
strat  = StrategyRegistry.load("mean_reversion_v3")

engine.add_strategy(strat, params={
    "lookback": 20,
    "z_entry":  2.0,
    "z_exit":   0.5,
})

report = engine.run(start="2023-01-01", end="2024-12-31")

# equity_curve is a polars.DataFrame
print(f"sharpe={report.sharpe:.2f} max_dd={report.max_dd:.2%}")
# → sharpe=1.87 max_dd=-6.42%

// Act III — DigiSearch

Find the right document. Ground any answer.

// Selective ingest, chunking,
// embedding, vector recall.
// Multi-mode retrieval over a
// pluggable store registry.

  • // Hybrid BM25 + dense vector
  • // Re-ranking via cross-encoder
  • // Citations attached to every result
from digisearch import Index, Document

idx = Index(store="qdrant", embed="text-embedding-3-small")

for path in pathlib.Path("./docs").rglob("*.md"):
    idx.upsert(Document.from_path(path))

results = idx.query(
    "how does the supervisor dispatch tools?",
    k=8, mode="hybrid", rerank=True,
)

for r in results:
    print(f"[{r.score:.3f}]", r.chunk.source, "→", r.chunk.preview)
# [0.913] digigraph/orchestration/supervisor.py → registry.discover()
# [0.881] docs/ARCHITECTURE.md              → tool dispatch flow

// Act IV — DigiChat

Talk to your stack. Ship with your own keys.

// Next.js BFF over DigiGraph.
// BYOK via LiteLLM translation.
// Streaming responses, scoped
// JWTs adapt the UI per user.

  • // Auth.js + Drizzle + AI SDK
  • // Scopes gate the tool palette
  • // Audit trail on every message
import { streamText } from 'ai';
import { digigraph } from '@digithings/sdk';

export async function POST(req: Request) {
  const { messages, session } = await req.json();

  const result = await streamText({
    model: digigraph({ baseURL: 'https://graph.local/v1' }),
    messages,
    tools: session.scopes,   // adaptive surface
    headers: { Authorization: `Bearer ${session.jwt}` },
  });

  return result.toDataStreamResponse();
}
